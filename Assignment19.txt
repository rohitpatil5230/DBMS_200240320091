/*1) Create a view that shows all of the customers who have the highest ratings.*/
create view rating_view as select * from customers 
where rating=(select max(rating) from customers);

/*2) Create a view that shows the number of salespeople in each city.*/
create view salespeople_view as select city,count(city) from salespeople group by city;

/*3) Create a view that shows the average and total orders for each salesperson after
his or her name. Assume all names are unique.*/
create view orders_view as select sname,avg(amt),count(onum) from salespeople,orders 
where orders.snum=salespeople.snum group by sname;

/*4) Create a view that shows each salesperson with multiple customers.*/
create view customers_salespeople_view as select sname,cname from customers, salespeople 
where customers.snum=salespeople.snum ;
